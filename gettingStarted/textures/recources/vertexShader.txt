#version 330 core
layout (location = 0) in vec3 aPos;
layout (location = 1) in vec3 aColor;
layout (location = 2) in vec2 aTexture;

out vec3 cPos;
out vec3 ourColor;
out vec2 TexCoord;

uniform float x_offset;

void main() {
   vec3 pos = aPos;
   pos.y *= -1;
   pos.x += x_offset;
   gl_Position = vec4(pos, 1.0);
   
   cPos = pos;

   ourColor = aColor;
   TexCoord = aTexture;
}
